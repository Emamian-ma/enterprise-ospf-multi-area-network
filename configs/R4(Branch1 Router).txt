enable
configure terminal

hostname R4
no ip domain-lookup
service password-encryption

!
! ===== Uplink to Access Switch (Router-on-a-Stick) =====
interface GigabitEthernet0/1
 no ip address
 no shutdown

interface GigabitEthernet0/1.10
 encapsulation dot1Q 10
 ip address 192.168.10.1 255.255.255.0

interface GigabitEthernet0/1.20
 encapsulation dot1Q 20
 ip address 192.168.20.1 255.255.255.0

!
! ===== WAN / OSPF Uplink to R2 =====
interface GigabitEthernet0/0
 ip address 10.0.24.2 255.255.255.252
 no shutdown

!
! ===== Loopback =====
interface Loopback0
 ip address 4.4.4.4 255.255.255.255

!
! ===== OSPF =====
router ospf 1
 router-id 4.4.4.4
 passive-interface GigabitEthernet0/1
 passive-interface GigabitEthernet0/1.10
 passive-interface GigabitEthernet0/1.20
 passive-interface Loopback0
 network 10.0.24.0 0.0.0.3 area 10
 network 192.168.10.0 0.0.0.255 area 10
 network 192.168.20.0 0.0.0.255 area 10
 network 4.4.4.4 0.0.0.0 area 10

end
write memory
